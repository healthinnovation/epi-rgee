The objective of this demo is to extract a time series of meteorological variables like temperature and precipitation in Lima using the dataset of Terraclim.

Information:
- For this demo you need to have rgee, sf, tidyverse ,lubridate, and patchwork packages previously installed.

2. Reading vector layer of study area

lima <- st_read(
  "https://github.com/ambarja/gpkg-pe/raw/main/Lima_provincia.gpkg",
  quiet = TRUE) %>%
  summarise()

3. Transformation of sf object to a feature collection

lima_ee <-  lima %>% 
  sf_as_ee()

4. Cooking data with rgee

terraclim <-  ee$ImageCollection$Dataset$IDAHO_EPSCOR_TERRACLIMATE$
  select(c("tmmx","pr"))$
  filterDate("1990-01-01","2021-12-31")$
  toBands()
# Extracting data
data <- ee_extract(
  x = terraclim,
  y = lima_ee,
  fun = ee$Reducer$mean(),
  sf = FALSE)

5. Cooking data for mapping

data_prev <- data %>% as_tibble() %>% 
  mutate(region = "LIMA") %>% 
  pivot_longer(X199001_pr:X202012_tmmx) %>% 
  separate(col = name,into = c("date","variable"),sep = "_") %>% 
  mutate(date = ym(gsub("X","",date))) %>% 
  separate(col = date,into = c("year","month"),sep = "-") %>% 
  mutate(month = factor(month,labels = month.abb))

6. Time series plot

data_prev %>% 
  filter(variable == "tmmx") %>% 
  ggplot(aes(x = as.integer(year),y = value*0.1, group = year, fill =  year)) + 
  geom_boxplot(color = "black",lwd=0.5,alpha=0.8) + 
  theme_minimal(8) + 
  theme(legend.position = "none") + 
  scale_fill_viridis_d(option = "inferno") +
  labs(x = "", y = "Temp")

7. Plots of temperature and precipitation

newdata <- data_prev %>% 
  group_by(month,variable) %>% 
  summarise(value = mean(value,na.rm = TRUE)) %>% 
  spread(variable,value) %>% 
  rename(temp = pr,
         pp = tmmx) 

pp <- ggplot(newdata,aes(x = month,y = pp)) +
  geom_bar(stat = "identity",fill = "blue",alpha=0.5) +
  theme_minimal() +
  labs(x = "",
       y = "Precipitation [mm]",
       caption = "Source: TerraClimate")

temp <- ggplot(newdata,aes(x = month,y = temp,group = 1)) + 
  geom_line(color = "red") + 
  theme_minimal() + 
  labs(
    x = "",
    y = "Temperature[C°]"
  )
